ZeroAuth SDK

Passwordless verification for any website

Demo: Built-in Button + Modal
Code Example
// 1. Configure (before importing)
window.ZeroAuthConfig = {
  relayUrl: 'https://zeroauth-relay.onrender.com'
};

// 2. Add button
import { ZeroAuthButton } from 'https://unpkg.com/@zero-auth/sdk/dist/zero-auth-sdk.umd.js';

document.querySelector('#app').innerHTML = ZeroAuthButton({
  text: 'Sign in with ZeroAuth',
  credentialType: 'Age Verification',
  claims: ['birth_year'],
  onSuccess: (result) => console.log('Verified!', result),
  onError: (error) => console.error('Error:', error)
});
