ZeroAuth SDK

Passwordless ZK verification - Just add script!

Method 1: Drop-in (no npm, no build)
Method 2: Trigger on your own button
Copy & Paste this code:
<!-- 1. Add this script (that's it!) -->
<script src="https://unpkg.com/@zero-auth/sdk/dist/zero-auth-sdk.umd.js"></script>

<!-- 2. Configure (optional, can also pass in init) -->
<script>
  ZeroAuth.init({
    relayUrl: 'https://your-relay.com',
    verifierName: 'Your App'
  });
</script>

<!-- 3. Add button with za-verify attribute -->
<button class="za-verify" 
  data-credential-type="Age Verification"
  data-claims='["birth_year"]'
  data-use-case="LOGIN">
  Sign in with ZeroAuth
</button>
